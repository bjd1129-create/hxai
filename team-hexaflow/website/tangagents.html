💻 快速开始

Quick Start - 5 分钟创建你的第一个 Agent 团队

from tangagents import Agent, Team, Task # 创建 Agent（融合 CrewAI 角色 + OpenClaw 工具） researcher = Agent( role="AI 研究员", goal="完成 AI 技术调研", tools=["web_search", "file_read", "code_exec"], model="minimax-m1" ) writer = Agent( role="技术作家", goal="撰写技术报告", tools=["file_write", "markdown_format"], model="minimax-m1" ) # 组建团队 team = Team([researcher, writer]) # 创建任务 task = Task( description="调研最新 AI 论文并生成报告", expected_output="Markdown 格式的调研报告", output_file="report.md" ) # 执行任务 result = team.run(task) print(f"报告已生成：{result.output_file}")
